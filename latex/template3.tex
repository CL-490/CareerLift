\newcommand{\resumeItem}[2]{%
  \item{%
    \textbf{#1}{:\hspace{0.5mm}#2 \vspace{-0.5mm}}%
  }%
}

\newcommand{\resumePOR}[3]{%
\vspace{0.5mm}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
        \textbf{#1},\hspace{0.3mm}#2 & \textit{\small{#3}} 
    \end{tabular*}
    \vspace{-2mm}
}

\newcommand{\resumeSubheading}[4]{%
\vspace{0.5mm}\item
    \begin{tabular*}{0.98\textwidth}[t]{l@{\extracolsep{\fill}}r}
        \textbf{#1} & \textit{\footnotesize{#4}} \\
        \textit{\footnotesize{#3}} &  \footnotesize{#2}\\
    \end{tabular*}
    \vspace{-2.4mm}
}

\newcommand{\resumeProject}[4]{%
\vspace{0.5mm}\item
    \begin{tabular*}{0.98\textwidth}[t]{l@{\extracolsep{\fill}}r}
        \textbf{#1} & \textit{\footnotesize{#3}} \\
        \footnotesize{\textit{#2}} & \footnotesize{#4}
    \end{tabular*}
    \vspace{-2.4mm}
}

\newcommand{\resumeSubItem}[2]{\resumeItem{#1}{#2}\vspace{-4pt}}

\renewcommand{\labelitemi}{$\vcenter{\hbox{\tiny$\bullet$}}$}

\newcommand{\resumeSubHeadingListStart}{\begin{itemize}[leftmargin=*,labelsep=0mm]}
\newcommand{\resumeHeadingSkillStart}{\begin{itemize}[leftmargin=*,itemsep=1.7mm, rightmargin=2ex]}
\newcommand{\resumeItemListStart}{\begin{justify}\begin{itemize}[leftmargin=3ex, rightmargin=2ex, noitemsep,labelsep=1.2mm,itemsep=0mm]\small}

\newcommand{\resumeSubHeadingListEnd}{\end{itemize}\vspace{2mm}}
\newcommand{\resumeHeadingSkillEnd}{\end{itemize}\vspace{-2mm}}
\newcommand{\resumeItemListEnd}{\end{itemize}\end{justify}\vspace{-2mm}}

% Education row in tabular
%   \education{Year}{Degree/Certificate}{Institute}{CGPA/Percentage}
\newcommand{\education}[4]{%
  \AppendTo{\EducationTableRows}{%
    #1 & #2 & #3 & #4\\\hline
  }%
}

% Experience entry
%   \experience{Company}{Location}{Position}{Dates}{items}
% where {items} is a block of \item ... lines (no begin/end{itemize})
\newcommand{\experience}[5]{%
  \AppendTo{\ExperienceList}{%
    \resumeSubheading{#1}{#2}{#3}{#4}%
    \resumeItemListStart
      #5
    \resumeItemListEnd
  }%
}

% Project entry
%   \project{Project Name}{Course/Context}{Dates}{Right side (e.g. link)}{items}
\newcommand{\project}[5]{%
  \AppendTo{\ProjectsList}{%
    \resumeProject{#1}{#2}{#3}{#4}%
    \resumeItemListStart
      #5
    \resumeItemListEnd
  }%
}

% Technical skills section rows
%   \techskill{Category}{Text}
\newcommand{\techskill}[2]{%
  \AppendTo{\TechSkillsList}{%
    \resumeSubItem{#1}{#2}%
  }%
}

% Key courses section rows
%   \keycourses{Category}{Text}
\newcommand{\keycourses}[2]{%
  \AppendTo{\KeyCoursesList}{%
    \resumeSubItem{#1}{#2}%
  }%
}

% Certifications
%   \certification{URL}{Certificate Name}{Institution}
\newcommand{\certification}[3]{%
  \AppendTo{\CertificationsList}{%
    \item{\href{#1}{\textbf{#2}} \hspace{1mm} #3}%
  }%
}

% Positions of responsibility
%   \position{Title}{Org}{Year}
\newcommand{\position}[3]{%
  \AppendTo{\PositionsList}{%
    \resumePOR{#1}{#2}{#3}%
  }%
}

% Misc achievements
%   \miscitem{Title}{Detail}{Year}
\newcommand{\miscitem}[3]{%
  \AppendTo{\MiscList}{%
    \resumePOR{#1}{#2}{#3}%
  }%
}

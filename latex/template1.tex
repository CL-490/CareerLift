% Award: \award{title}{description}
\newcommand{\award}[2]{%
  \g@addto@macro\AwardsList{%
    \item \textbf{#1}: #2%
  }%
}

% Experience: \experience{title}{organization}{dates}{bullet1}{bullet2}{...}
\newcommand{\experience}[3]{%
  \g@addto@macro\ExperienceList{%
    \textbf{#1} $|$ #2 \hfill #3\\
    \begin{itemize}
  }%
}

% Experience bullet
\newcommand{\experienceBullet}[1]{%
  \g@addto@macro\ExperienceList{%
    \item #1%
  }%
}

% Close experience
\newcommand{\experienceEnd}{%
  \g@addto@macro\ExperienceList{%
    \end{itemize}
    \vspace{0.5em}
  }%
}

% Project: \project{title}{url}{bullet1}{bullet2}{...}
\newcommand{\project}[2]{%
  \g@addto@macro\ProjectsList{%
    \if\relax\detokenize{#2}\relax
      \textbf{#1}\\
    \else
      \textbf{\href{#2}{#1}}\\
    \fi
    \begin{itemize}[leftmargin=*]
  }%
}

% Project bullet
\newcommand{\projectBullet}[1]{%
  \g@addto@macro\ProjectsList{%
    \item #1%
  }%
}

% Close project
\newcommand{\projectEnd}{%
  \g@addto@macro\ProjectsList{%
    \end{itemize}
    \vspace{0.3em}
  }%
}

% Leadership: \leadership{title}{dates}{bullet1}{bullet2}{...}
\newcommand{\leadership}[2]{%

    \begin{itemize}
  }%

% Leadership bullet

  }%
}

% Close leadership
\newcommand{\leadershipEnd}{%

    \end{itemize}
    \vspace{0.3em}
  }%
}

% Education entry
% \education{Years}{Degree}{Institution}{Location}{Details block}
% {Details block} = plain text with \\ line breaks as needed.
\newcommand{\education}[5]{%
  \AppendTo{\EducationList}{%
    \blueitem{#1: #2}\\
    \textit{#3, #4}\\
    #5
    \vspace{0.3em}\\
  }%
}

% Experience entry
% \experience{Dates}{Title}{Org}{Location}{items}
% {items} = \item ... \item ...
\newcommand{\experience}[5]{%
  \AppendTo{\ExperienceList}{%
    \blueitem{#1: #2}\\
    \textit{#3, #4}
    \begin{itemize}
      #5
    \end{itemize}
    \vspace{0.3em}
  }%
}

% Publication entry
% \publication{Title}{Venue/Status}{Authors line}
\newcommand{\publication}[3]{%
  \AppendTo{\PublicationList}{%
    \blueitem{#1} #2\\
    \textit{#3}\\[0.2em]
  }%
}

% Project entry
% \project{Title}{Context}{items}
% {items} = \item ...
\newcommand{\project}[3]{%
  \AppendTo{\ProjectList}{%
    \blueitem{#1}\\
    \textit{#2}
    \begin{itemize}
      #3
    \end{itemize}
    \vspace{0.3em}
  }%
}

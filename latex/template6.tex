% Summary
\newcommand{\summaryitem}[1]{%
  \AppendTo{\SummaryList}{%
    \item \footnotesize #1%
  }%
}

% Education
% \education{Level}{Institution}{Marks}{RightText (e.g. Years)}
\newcommand{\education}[4]{%
  \AppendTo{\EducationList}{%
    \item
    \begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
      \textbf{#1} & #4\\
    \end{tabular*}\\[-0.1cm]
    {#2}\\
    {Marks/CGPA: #3}%
  }%
}

% Skill category
% \skillcat{Category}{Text}
\newcommand{\skillcat}[2]{%
  \AppendTo{\SkillsList}{%
    \item \textbf{#1}\\[-0.05cm]
    {\footnotesize #2}%
  }%
}

% Project
% \project{Title}{LeftSubtitle}{RightInfo (e.g. dates / "Valid through ...")}{DemoURL}{RepoURL}
\newcommand{\project}[5]{%
  \AppendTo{\ProjectsList}{%
    \item
    \begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
      \textbf{#1} & {\footnotesize #3}%
    \end{tabular*}\\[-0.05cm]
    {\footnotesize #2}\\[-0.1cm]
    {\footnotesize
      \ifthenelse{\equal{#4}{}}{}{ \href{#4}{Demo}}
      \ifthenelse{\equal{#4#5}{}}{}{\hspace{1.5em}}
      \ifthenelse{\equal{#5}{}}{}{ \href{#5}{GitHub}}
    }%
  }%
}

% Significant role
% \role{RoleName}{Org/location}{RightInfo (dates)}
\newcommand{\role}[3]{%
  \AppendTo{\RolesList}{%
    \item
    \begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
      \textbf{#1} & #3\\
    \end{tabular*}\\[-0.1cm]
    {#2}%
  }%
}

% Experience
% \experience{Title}{Org/location}{RightInfo (dates)}{items = \item ...}
\newcommand{\experience}[4]{%
  \AppendTo{\ExperienceList}{%
    \item
    \begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
      \textbf{#1} & #3\\
    \end{tabular*}\\[-0.1cm]
    {#2}%
    \begin{itemize}[noitemsep,nolistsep]
      #4
    \end{itemize}
  }%
}

% Global certification
% \certification{Name}{RightInfo (e.g. "Valid through 2027")}{ExtraLine (optional)}
\newcommand{\certification}[3]{%
  \AppendTo{\CertificationsList}{%
    \item
    \begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
      \textbf{#1} & {\footnotesize #2}%
    \end{tabular*}%
    \ifthenelse{\equal{#3}{}}{}{{\\[-0.05cm]{\footnotesize #3}}}%
  }%
}

% Reference
% \refitem{Name}{Designation}{RightInfo (e.g. Contact link)}
\newcommand{\refitem}[3]{%
  \AppendTo{\ReferencesList}{%
    \item
    \begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
      \textbf{#1} & {\footnotesize #3}%
    \end{tabular*}\\[-0.1cm]
    {\footnotesize #2}%
  }%
}

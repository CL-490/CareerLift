% Summary bullet
% \summaryitem{text}
\newcommand{\summaryitem}[1]{%
  \AppendTo{\SummaryList}{%
    \item \footnotesize #1%
  }%
}

% Education table row
% \educationrow{Course}{College}{Year}{CGPA}
\newcommand{\educationrow}[4]{%
  \AppendTo{\EducationTableRows}{%
    #1 & #2 & #3 & #4\\
  }%
}

% Achievement
% \achievement{text}{year in [YYYY]}
\newcommand{\achievement}[2]{%
  \AppendTo{\AchievementsList}{%
    \item #1 \hfill {#2}%
  }%
}

% Experience
% \experience{Role | Org}{Period}{items=\item ...}
\newcommand{\experience}[3]{%
  \AppendTo{\ExperienceList}{%
    \item \textbf{#1}\hfill {#2}\par
    \begin{itemize}[noitemsep,nolistsep]
      #3
    \end{itemize}
  }%
}

% Skill category
% \skillcat{Label}{Text}
\newcommand{\skillcat}[2]{%
  \AppendTo{\SkillsList}{%
    \item \textbf{#1}: #2
  }%
}

% Project
% \project{Title | Instructor}{Period}{DemoURL}{RepoURL}{items=\item ...}
\newcommand{\project}[5]{%
  \AppendTo{\ProjectsList}{%
    \item \textbf{#1}\hfill {#2}\par
    {\footnotesize
      \hspace{2em}%
      \if\relax\detokenize{#3}\relax\else\href{#3}{Link}\fi
      \hspace{2em}%
      \if\relax\detokenize{#4}\relax\else\href{#4}{Github}\fi
    }\par
    \begin{itemize}[noitemsep,nolistsep]
      #5
    \end{itemize}
  }%
}

% Courses block
% \coursesblock{Label}{Text}{Tag}
\newcommand{\coursesblock}[3]{%
  \AppendTo{\CoursesList}{%
    \item \textbf{#1}\hfill {#3}\par
    \textbf{#2}\par
  }%
}

% Position of responsibility
% \position{Title | Body}{Period}{items=\item ...}
\newcommand{\position}[3]{%
  \AppendTo{\PositionsList}{%
    \item \textbf{#1}\hfill {#2}\par
    \begin{itemize}[noitemsep,nolistsep]
      #3
    \end{itemize}
  }%
}

% Extracurricular

    \item #1%
  }%
}
